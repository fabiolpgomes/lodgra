## Dev Notes

### Schema proposto

```sql
-- Catálogo de comodidades
CREATE TABLE amenities (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text NOT NULL,
  icon text NOT NULL,           -- nome do ícone Lucide (ex: 'wifi', 'tv', 'car')
  category text NOT NULL,       -- 'sala', 'quarto', 'cozinha', 'banheiro', 'seguranca', 'destaque', 'geral'
  sort_order integer DEFAULT 0
);

-- Relação propriedade ↔ comodidades
CREATE TABLE property_amenities (
  property_id uuid NOT NULL REFERENCES properties(id) ON DELETE CASCADE,
  amenity_id uuid NOT NULL REFERENCES amenities(id) ON DELETE CASCADE,
  PRIMARY KEY (property_id, amenity_id)
);

-- Quartos
CREATE TABLE property_rooms (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id uuid NOT NULL REFERENCES properties(id) ON DELETE CASCADE,
  name text,                    -- ex: "Quarto Principal"
  bed_type text NOT NULL,       -- 'single', 'double', 'queen', 'king', 'sofa_bed', 'bunk'
  bed_count integer NOT NULL DEFAULT 1,
  provides_linen boolean NOT NULL DEFAULT false,
  sort_order integer DEFAULT 0
);

-- Banheiros
CREATE TABLE property_bathrooms (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  property_id uuid NOT NULL REFERENCES properties(id) ON DELETE CASCADE,
  name text,                    -- ex: "Casa de Banho 1"
  bathroom_type text NOT NULL,  -- 'wc', 'full'
  amenities text[],             -- ['towels', 'hairdryer', 'soap', 'shampoo', ...]
  sort_order integer DEFAULT 0
);

-- Taxas e horários (adicionar à tabela properties)
ALTER TABLE properties
  ADD COLUMN cleaning_fee numeric,
  ADD COLUMN cleaning_fee_type text,    -- 'per_stay', 'per_night'
  ADD COLUMN pet_fee numeric,
  ADD COLUMN pet_fee_type text,         -- 'per_stay', 'per_night'
  ADD COLUMN checkin_from time,
  ADD COLUMN checkin_until time,
  ADD COLUMN checkout_until time;
```

### Seed do catálogo de comodidades

Criar seed com comodidades comuns por categoria (WiFi, TV, AC, etc.) com ícones Lucide.
