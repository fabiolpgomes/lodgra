## Story

**Como** visitante na página pública de uma propriedade,
**Quero** ver o score médio das avaliações agrupado por OTA,
**Para que** possa aferir a credibilidade do alojamento com base em fontes que já conheço.

```typescript
  source: ReviewSource
  avg: number      // média arredondada a 1 decimal
  count: number
}

export type ReviewScoreData = {
  globalAvg: number   // média global arredondada a 1 decimal
  totalCount: number
  bySource: OtaScore[]
}
```

### Cálculo do agregado (server-side em `/p/[slug]/page.tsx`)

```typescript
const { data: reviewsRaw } = await adminClient
  .from('property_reviews')
  .select('source, rating')
  .eq('property_id', property.id)

const reviews = reviewsRaw ?? []
let reviewScore: ReviewScoreData | null = null

if (reviews.length > 0) {
  const globalAvg = Math.round((reviews.reduce((s, r) => s + Number(r.rating), 0) / reviews.length) * 10) / 10

  const bySourceMap = new Map<string, number[]>()
  for (const r of reviews) {
    if (!bySourceMap.has(r.source)) bySourceMap.set(r.source, [])
    bySourceMap.get(r.source)!.push(Number(r.rating))
  }

  const bySource = Array.from(bySourceMap.entries()).map(([source, ratings]) => ({
    source: source as ReviewSource,
    avg: Math.round((ratings.reduce((s, r) => s + r, 0) / ratings.length) * 10) / 10,
    count: ratings.length,
  }))

  reviewScore = { globalAvg, totalCount: reviews.length, bySource }
}
```

### Labels qualitativos

```typescript
function qualityLabel(avg: number): string | null {
  if (avg >= 9) return 'Excelente'
  if (avg >= 8) return 'Muito Bom'
  if (avg >= 7) return 'Bom'
  if (avg >= 6) return 'Satisfatório'
  return null
}
```
